\section{The \JoinCalc}

\begin{JDef}{grammar of the \joincalc}
  P, Q
  \grmr \jsnd{x}{v_1, \dots, v_n}
  \altn \jdef{D}{P}
  \altn P \jpar Q
  \altn \jnullproc
  \\ \\
  D, E
  \grmr \jrct{J}{P}
  \altn D \jcon E
  \altn \jnulldef
  \\ \\
  J, K
  \grmr \jrcv{x}{y_1, \dots, y_n}
  \altn J \jpat K
\end{JDef}

\begin{JDef}{grammar of the monadic \joincalc}
  P, Q
  \grmr \jsnd{x}{u}
  \altn P \jpar Q
  \altn \jdef{\jrct{\jrcv{x}{u} \jpat \jrcv{y}{v}}{P}}{Q}
\end{JDef}
