\section{The Pi Calculus}

\subsection{Monadic}

\begin{definition}{grammar of the \picalc}
  P, Q
  \grmr \Sigma_{i \in I} \pi_i . P_i
  \altn P \para Q
  \altn !P
  \altn \new{x} P
  \\ \\
  \pi
  \grmr \recv{x}{y}
  \alt  \send{x}{y}
\end{definition}

\begin{definition}{bound and free names}
  \bn(\recv{x}{y}) &= \{y\} \\
  \bn(\send{x}{y}) &= \emptyset \\
  \fn(\recv{x}{y}) &= \{x\} \\
  \fn(\send{x}{y}) &= \{x,y\} \\
  \names(P) &= \bn(P) \cup \fn(P)
\end{definition}

    {\send{x}{y_1 \dotsb y_n}}
